Description

Set configuration for the components

Example

import {ThemeProvider, createTheme} from '@mui/material'
import Muitils from 'muitils'

const theme = createTheme()

Muitils.configure({
  Form: {
    sx: {
      classes: {
        'form_input_row': {
          color: 'red'
        }
      }
    },
    transform: (component) => (
     <ThemeProvider theme={theme}>
        {component}
      </ThemeProvider>
    ),
    props: {
      className: 'test'
    }
  }
})
